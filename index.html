Setup Instructions
1. Open Google Sheets → Create new sheet named "FitLog"
2. Add sheets: Calories, Workouts, Weight
3. Go to Extensions → Apps Script
4. Paste the script below and deploy as Web App
5. Copy the URL and paste above 
function doGet(e){
  var sheet = e.parameter.sheet;
  var row = JSON.parse(e.parameter.row);
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sh = ss.getSheetByName(sheet);
  if (!sh) {
    sh = ss.insertSheet(sheet);
    // Add headers based on sheet type
    if (sheet === 'Calories') {
      sh.appendRow(['Date','Time','Food','Calories','Protein (g)','Carbs (g)','Meal']);
      sh.getRange(1,1,1,7).setFontWeight('bold').setBackground('#1a1a2e').setFontColor('#6ef08b');
    } else if (sheet === 'Daily Summary') {
      sh.appendRow(['Date','Total Cal','Total Protein (g)','Total Carbs (g)','Goal (kcal)','Balance','Items']);
      sh.getRange(1,1,1,7).setFontWeight('bold').setBackground('#1a1a2e').setFontColor('#6ef08b');
    } else if (sheet === 'Workouts') {
      sh.appendRow(['Date','Muscles','Exercises','Calories Burned','Summary']);
      sh.getRange(1,1,1,5).setFontWeight('bold').setBackground('#1a1a2e').setFontColor('#6ef08b');
    } else if (sheet === 'Weight') {
      sh.appendRow(['Date','Weight (kg)']);
      sh.getRange(1,1,1,2).setFontWeight('bold').setBackground('#1a1a2e').setFontColor('#6ef08b');
    }
  }
  // For Daily Summary, update today's row if exists instead of appending
  if (sheet === 'Daily Summary' && sh.getLastRow() > 1) {
    var data = sh.getDataRange().getValues();
    var today = row[0];
    for (var i = 1; i < data.length; i++) {
      if (data[i][0] == today) {
        sh.getRange(i+1, 1, 1, row.length).setValues([row]);
        return ContentService.createTextOutput(JSON.stringify({ok:true,updated:true}))
          .setMimeType(ContentService.MimeType.JSON);
      }
    }
  }
  sh.appendRow(row);
  // Auto-resize columns
  sh.autoResizeColumns(1, sh.getLastColumn());
  return ContentService.createTextOutput(JSON.stringify({ok:true}))
    .setMimeType(ContentService.MimeType.JSON);
}
